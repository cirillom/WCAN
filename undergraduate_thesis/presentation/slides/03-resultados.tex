\begin{frame}{A Campanha Experimental}
    \begin{itemize}
        \item \textbf{Base de Dados:} 679 execuções úteis e reprodutíveis (10 Hz a 1000 Hz).
        \item \textbf{Eixos de Análise:} O desempenho do protocolo foi avaliado sob duas óticas principais:
        \begin{enumerate}
            \item \textbf{Eixo 1 (Mecanismo):} O impacto do Agrupamento Temporal (\textit{Batching}).
            \item \textbf{Eixo 2 (Transporte):} O \textit{Trade-off} entre \textit{Broadcast} e \textit{Multicast} sob alta carga.
        \end{enumerate}
    \end{itemize}
\end{frame}

% SCRIPT: Focar no Eixo 1. Mostrar a tabela do efeito do agrupamento.
% Explicar por que sem lotes o desempenho cai abruptamente: a taxa de pacote iguala a taxa de amostra, saturando o driver Wi-Fi.
\begin{frame}{Eixo 1: O Efeito do Agrupamento Temporal (\textit{Batching})}
    
    \begin{table}[]
        \centering
        \small
        \renewcommand{\arraystretch}{1.3}
        \begin{tabular}{l | c | p{5.5cm}}
            \hline
            \textbf{Configuração} & \textbf{Aprovadas} & \textbf{Interpretação} \\
            \hline
            \textbf{Com Lotes} (\texttt{linger\_ms = 100}) & \textbf{415 / 499} (83\%) & O pacote agrega amostras, amortizando o custo de rádio, \textit{callback} e confirmação. \\
            \hline
            \textbf{Sem Lotes} (\texttt{linger\_ms = 0}) & \textbf{38 / 180} (21\%) & Envio amostra a amostra satura filas e \textit{driver} ESP-NOW (\texttt{ESP\_ERR\_ESPNOW\_INTERNAL}). \\
            \hline
        \end{tabular}
    \end{table}

    \vspace{0.4cm}
    \textbf{Conclusão do Eixo 1:}
    Para sustentar alta frequência (até 1000 Hz), a formação de lotes não é opcional; ela é o mecanismo vital que impede a saturação do caminho crítico.
\end{frame}

\begin{frame}{Eixo 2: O \textit{Trade-off} dos Transportes sob Carga}
    \textbf{Cenário de Teste (\texttt{multiple}):} 3 identificadores CAN por sensor (simulando um nó real).
    
    \vspace{0.2cm}
    \begin{table}[]
        \centering
        \small
        \renewcommand{\arraystretch}{1.3}
        \begin{tabular}{l | c | c | c}
            \hline
            \textbf{Transporte} & \textbf{750--1000 Hz} & \textbf{Total da Suíte} & \textbf{Taxa de Entrega} \\
            \hline
            \textit{\textbf{Multicast}} & \textbf{36 / 36} & \textbf{90 / 90} & \textbf{100,0\%} \\
            \hline
            \textit{Broadcast} & 3 / 36 & 34 / 90 & 37,8\% \\
            \hline
        \end{tabular}
    \end{table}

    \vspace{0.4cm}
    \textbf{O Gargalo do \textit{Broadcast}:} 
    \begin{itemize}
        \item Aguardar o ACK da aplicação para \textbf{cada lote} e realizar retransmissões esgota o tempo de fila. Em 1000 Hz (4S-1R), a fila chegou a atrasos de \textbf{3,83 s}.
    \end{itemize}
    \textbf{A Vantagem do \textit{Multicast}:}
    \begin{itemize}
        \item Dispensa o ACK de aplicação lote a lote. A confirmação no nível MAC do enlace 802.11 reduz a coordenação e preserva a vazão.
    \end{itemize}
\end{frame}
