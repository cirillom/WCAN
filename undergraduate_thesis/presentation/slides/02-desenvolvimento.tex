% Depois explicar por que não Wi-Fi ou Bluetooth e sim ESP-NOW.
\begin{frame}{Escolha da Plataforma}
    \begin{itemize}
        \item \textbf{O Padrão da Equipe (Família STM32):} 
        \begin{itemize}
            \item Foi a primeira opção considerada, por ser o ecossistema já dominado pela equipe. 
            \item \textbf{Descarte:} Linhas integradas (como STM32WB) suportam BLE/802.15.4, mas não Wi-Fi nativo. Usar rádio externo aumentaria o tamanho e a complexidade do módulo sensor.
        \end{itemize}
        
        \item \textbf{A Escolha Final (ESP32):} Rádio integrado, baixo custo (R\$40--70), programação em C/C++ (FreeRTOS) e placa extremamente compacta (ex: Beetle ESP32-C3).
        
        \item \textbf{Protocolo de Transporte (Por que ESP-NOW?):} 
        \begin{itemize}
            \item \textbf{vs BLE:} BLE exige estado de conexão e descoberta lenta.
            \item \textbf{vs Wi-Fi:} Wi-Fi exige infraestrutura (IP/Roteador).
            \item \textbf{ESP-NOW:} Acesso direto à camada MAC, início imediato sem \textit{handshake} e adequado para \textit{soft real-time}.
        \end{itemize}
    \end{itemize}
\end{frame}

% Contrapor a semântica "segura e lenta" do broadcast vs a "rápida e MAC-level" do multicast.
\begin{frame}{Variantes de Transporte}
    \vspace{0.2cm}
    \textbf{\large 1. Variante \textit{Broadcast} (Primeiro ACK)}
    \begin{itemize}
        \item Confirmação explícita no nível da \textbf{aplicação} (via mensagem de controle).
        \item Tarefa dedicada (\textit{retry\_task}) gerencia \textit{timeouts} e retransmissões.
        \item Semântica segura, porém gera maior pressão sobre as filas em alta carga.
    \end{itemize}
    
    \vspace{0.6cm}
    \textbf{\large 2. Variante \textit{Multicast} (Híbrido)}
    \begin{itemize}
        \item Receptores anunciam interesse por IDs CAN específicos ("assinam").
        \item Confirmação de entrega baseada no nível \textbf{MAC} do IEEE 802.11 (via \textit{callback}).
        \item Menor latência no caminho crítico: dispensa o ACK de aplicação lote a lote.
    \end{itemize}
\end{frame}

% SCRIPT: O núcleo do software. Mostrar que as interrupções de rádio não fazem o trabalho pesado.
\begin{frame}{Arquitetura de Software: Núcleo \texttt{TransceiverBase}}
    \begin{block}{Abordagem FreeRTOS}
        Separação do fluxo crítico em tarefas isoladas e uso de \textit{callbacks} curtos, deslocando o trabalho pesado para \textit{tasks} de menor acoplamento temporal.
    \end{block}
    \begin{itemize}
        \item \textbf{Comunicação Inter-tarefas:} Uso de filas (\texttt{Queue}) do FreeRTOS.
        \item \textbf{Anéis de Memória (\texttt{RingBuffer}):} Absorção de picos (\textit{bursts}) de mensagens na transmissão e recepção, organizados por ID CAN.
        \item \textbf{Deduplicação (RX):} Filtro baseado em MAC de origem, ID CAN e \texttt{sequence\_id} para ignorar retransmissões redundantes do rádio.
    \end{itemize}
\end{frame}

\begin{frame}{Automação da Campanha de Testes}
    \begin{itemize}
        \item \textbf{Orquestração via Python:} Uso de um \textit{runner} automatizado no computador (\textit{host}) para enviar comandos às placas via UART.
        \item \textbf{Configuração YAML:} Suítes, topologias (até 5 placas simultâneas), transportes e frequências definidas em arquivos estruturados.
        \item \textbf{Sincronização Segura:} Protocolo \textit{handshake} (WAIT $\rightarrow$ CONFIG $\rightarrow$ ACK $\rightarrow$ START) garantindo que os nós iniciassem os ensaios rigorosamente ao mesmo tempo.
        \item \textbf{Análise Final:} \textit{Scripts} dedicados processaram as saídas seriais, reconstruindo os \textit{ranges} (contadores) recebidos versus os gerados.
    \end{itemize}
\end{frame}
