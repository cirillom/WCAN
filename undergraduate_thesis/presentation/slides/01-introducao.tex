% SCRIPT: Explicar o processo de decisão antes de chegar no WCAN.
% O cabeamento adiciona peso e os coletores rotativos (slip rings) são caros e frágeis.
% Módulos isolados com cartão SD resolvem o cabo, mas perdem a sincronização temporal com o resto do carro.
% A ponte sem fio (WCAN) foi o "sweet spot".
\begin{frame}{Soluções Consideradas}
    Antes do desenvolvimento do WCAN, três rotas arquiteturais foram avaliadas:
    
    \vspace{0.4cm}
    \begin{table}[]
        \centering
        \footnotesize
        \renewcommand{\arraystretch}{1.5}
        \begin{tabular}{p{0.25\textwidth} | p{0.32\textwidth} | p{0.35\textwidth}}
            \hline
            \textbf{Abordagem} & \textbf{Vantagens} & \textbf{Desvantagens} \\
            \hline
            \textbf{Cabeamento Dedicado} & Máxima confiabilidade elétrica. & Peso, fadiga mecânica e uso de \textit{slip rings} (rodas). \\
            \hline
            \textbf{Aquisição Independente} & Remove os chicotes elétricos curtos/longos. & Perde a visão em tempo real e dificulta sincronização temporal. \\
            \hline
            \textbf{Ponte Sem Fio (A Escolha)} & \textbf{Remove fios, integra ao sistema atual em tempo real.} & \textbf{Exige protocolo próprio e lida com falhas do meio aéreo.} \\
            \hline
        \end{tabular}
    \end{table}
\end{frame}

            \textbf{Objetivos Específicos:}
            \scriptsize
            \begin{itemize}
                \item Caracterizar os requisitos de uma ponte sem fio compatível com o sistema de aquisição existente;
                \item Implementar o WCAN como camada de comunicação sobre ESP-NOW;
                \item Desenvolver e comparar variantes de transporte adequadas;
                \item Avaliar, em bancada, o comportamento do sistema (sensores, filtros, frequências);
                \item Delimitar as condições de uso e pontos para validação no veículo.
            \end{itemize}
